\begin{homeworkProblem}
    Use the definition of $\theta$ in order to show the following:
    \begin{enumerate}
        \item $5n^3+2n^2+3n=\theta(n^3)$
        \item $\sqrt[2]{7n^2+2n-8}=\theta(n)$
    \end{enumerate}
    \textbf{Solution}
    \begin{enumerate}
        \item 
        \begin{proof}
            For the lower bound, when $C_1=4$, for $n\geq0$, \\
            $5n^3\leq5n^3+2n^2+3n$\\
            For the upper bound,\\
            $5n^3+2n^2+3n\leq6n^3$, when $C_2=6$, for $n\geq3$.\\
            $C_1\cdot g(n)\leq5n^3+2n^2+3n\leq C_2\cdot g(n)$\\
            $\leftrightarrow 4n^3\leq5n^3+2n^2+3n\leq 6n^3$\\
            $5n^3+2n^2+3n=\theta(n^3)$
        \end{proof}
        \item 
        \begin{proof}
            For $n\geq 4$, $\sqrt[2]{n^2}=\sqrt[2]{7}n\leq \sqrt[2]{7n^2+2n-8}$\\
            $C_1=\sqrt[2]{7}$\\
            For the upper bound, when $C_2=\sqrt[2]{7}+1$, for $n \geq 0$,\\
            $\sqrt[2]{7n^2+2n-8}\leq (\sqrt[2]{7}+1)n$\\
            $C_1\cdot g(n)\leq\sqrt[2]{7n^2+2n-8}\leq C_2\cdot g(n)$\\
            $\leftrightarrow \sqrt[2]{7}n\leq \sqrt[2]{7n^2+2n-8}\leq (\sqrt[2]{7}+1)n$\\
            $\sqrt[2]{7n^2+2n-8}=\theta(n)$
        \end{proof}
    \end{enumerate}
\end{homeworkProblem}
